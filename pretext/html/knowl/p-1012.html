Thus \begin{equation*} \prod_{j=2}^{k+1} \left( 1 - \frac{1}{j^2} \right) = \frac{k+1}{2k} \; \cdot \; \left( 1 - \frac{1}{(k+1)^2} \right) \end{equation*} \begin{equation*} = \frac{k+1}{2k} - \frac{(k+1)}{2k(k+1)^2} \end{equation*} \begin{equation*} = \frac{k+1}{2k} - \frac{(1)}{2k(k+1)} \end{equation*} \begin{equation*} = \frac{(k+1)^2 - 1}{2k(k+1)} \end{equation*}
